Add IPC handlers after existing listen handlers (around line 149):
```javascript
ipcMain.handle('listen:checkLameAvailable', async () => {
    const available = await recordingService.checkLameAvailable();
    return { available };
});

ipcMain.handle('listen:startRecording', async (event, { sessionId }) => {
    try {
        await recordingService.startRecording(sessionId);
        return { success: true };
    } catch (err) {
        return { success: false, error: err.message };
    }
});

ipcMain.handle('listen:sendAudioChunk', async (event, { data }) => {
    try {
        const pcmBuffer = Buffer.from(data, 'base64');
        await recordingService.writeAudioChunk(pcmBuffer);
        return { success: true };
    } catch (err) {
        return { success: false, error: err.message };
    }
});

ipcMain.handle('listen:stopRecording', async (event, { targetPath }) => {
    try {
        const result = await recordingService.stopAndSave(targetPath);
        return { success: true, path: result.path, fallback: result.fallback };
    } catch (err) {
        return { success: false, error: err.message };
    }
});
```
