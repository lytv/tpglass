mixAudio(micBuffer, systemBuffer) {
    // Both are Float32Array at 24kHz mono
    const length = Math.max(micBuffer.length, systemBuffer.length);
    const mixed = new Float32Array(length);

    // Simple mixing: add samples, clamp to [-1, 1]
    for (let i = 0; i < length; i++) {
        let sample = 0;
        if (i < micBuffer.length) sample += micBuffer[i];
        if (i < systemBuffer.length) sample += systemBuffer[i];
        mixed[i] = Math.max(-1, Math.min(1, sample)); // Soft clip
    }
    return mixed;
}
```

    if (!lameAvailable) {

    }
} catch (err) {
    console.warn('[Recording] Failed to start recording:', err.message);
}
```

// Save recording
try {
    const pathStore = require('electron-store');
    const pathConfig = new pathStore({ name: 'path-settings' });
    const lastSavePath = pathConfig.get('lastSavePath') || require('os').homedir();
    const timestamp = new Date().toISOString().slice(0, 16).replace(/[-:T]/g, '');
    const filename = `recording-${timestamp}.mp3`;
    const targetPath = require('path').join(lastSavePath, filename);

    const result = await recordingService.stopAndSave(targetPath);
    if (result.success) {
        console.log('[Recording] Saved to', result.path);
        if (result.fallback) {
            console.warn('[Recording] Saved as WAV due to encoding issues');
            if (window.showNotification) {
                window.showNotification('warning', 'Recording saved as WAV due to encoding issues');
            }
        }
    }
} catch (err) {

}
