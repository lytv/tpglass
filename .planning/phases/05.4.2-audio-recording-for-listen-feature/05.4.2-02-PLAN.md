2. In initializeSession() (around line 196, after sttReady check and before sending status):
```javascript
// Check lame and start recording
try {
    const lameAvailable = await recordingService.checkLameAvailable();
    if (!lameAvailable) {
        console.warn('[Recording] lame encoder not found, audio recording disabled');
    } else {
        await recordingService.startRecording(this.currentSessionId);
        console.log('[Recording] Started recording for session', this.currentSessionId);
    }
} catch (err) {
    console.warn('[Recording] Failed to start recording:', err.message);
}
```

3. In closeSession() (around line 237, after stopMacOSAudioCapture and before ending database session):
```javascript
// Save recording
try {
    const pathStore = require('electron-store');
    const pathConfig = new pathStore({ name: 'path-settings' });
    const lastSavePath = pathConfig.get('lastSavePath') || require('os').homedir();
    const timestamp = new Date().toISOString().slice(0, 16).replace(/[-:T]/g, '');
    const filename = `recording-${timestamp}.mp3`;
    const targetPath = require('path').join(lastSavePath, filename);

    const result = await recordingService.stopAndSave(targetPath);
    if (result.success) {
        console.log('[Recording] Saved to', result.path);
        if (result.fallback) {
            console.warn('[Recording] Saved as WAV due to encoding issues');
        }
    }
} catch (err) {
    console.warn('[Recording] Failed to save recording:', err.message);
}
```
